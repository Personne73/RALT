002import java.util.ArrayList;
003/**
004 * Cette classe permet de créer des pièces transporteuses qui déplacent le joueur aléatoirement dans une autre pièce
005 * 
006 * @author MBIAPA KETCHECKMEN Joël Trésor
007 * @version 2021.05
008 */
009public class pkg_room.pkg_room.pkg_room.TransporterRoom extends pkg_room.pkg_room.pkg_room.Room
010{
011    private pkg_room.pkg_room.pkg_room.RoomRandomizer aRandomRoom;
012    private ArrayList<pkg_room.pkg_room.pkg_room.Room> aRoomList;
013    private int aAleaInt;
014    
015    /**
016     * Constructeur qui permet d'initialiser les différents attributs
017     * @param pDescription la description de la pièce 
018     * @param pImageName le nom de l'image et son extension
019     * @param pRoomList la liste des pièces dans lesquels le joueur sera transporté
020     */
021    public pkg_room.pkg_room.pkg_room.TransporterRoom( final String pDescription, final String pImageName, final ArrayList pRoomList )
022    {
023        super(pDescription, pImageName);
024        this.aRoomList = pRoomList;
025        this.aRandomRoom = new pkg_room.pkg_room.pkg_room.RoomRandomizer(pRoomList);
026        this.aAleaInt = 6;
027    } // pkg_room.pkg_room.pkg_room.TransporterRoom(...)
028    
029    /**
030     * Accesseur pour récupérer une pièce aléatoirement
031     * @param pDirection la direction sous forme de chaine de caractère
032     * @return la pièce aléatoire qui sera utilisé pour transporter le joueur
033     */
034    public pkg_room.pkg_room.pkg_room.Room getExit( final String pDirection )
035    {
036        if ( this.aAleaInt == 6 ) return this.findRandomRoom();
037        else return this.aRoomList.get(this.aAleaInt);
038    } // getExit(.)
039    
040    /**
041     * Modificateur qui permet de modifier la valeur de l'attribut aAleaInt par l'entier passé en paramètre
042     * @param pAleaInt l'entier qui modifiera l'attribut
043     */
044    public void setAleaInt( final int pAleaInt )
045    {
046       this.aAleaInt =  pAleaInt;
047    } // setAleaInt(.)
048
049    /**
050     * Procédure privée qui permet d'avoir une pièce prise de manière aléatoire
051     * @return un pièce de façon random
052     */
053    private pkg_room.pkg_room.pkg_room.Room findRandomRoom()
054    {
055        return this.aRandomRoom.findRandomRoom();
056    } // findRandomRoom()
057    
058} // pkg_room.pkg_room.pkg_room.TransporterRoom
