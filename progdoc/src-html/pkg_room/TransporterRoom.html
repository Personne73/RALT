001package pkg_room;
002
003import java.util.ArrayList;
004/**
005 * Cette classe permet de créer des pièces transporteuses qui déplacent 
006 * le joueur aléatoirement dans une autre pièce
007 * 
008 * @author MBIAPA KETCHECKMEN Joël Trésor
009 * @version 2021.05
010 */
011public class TransporterRoom extends Room
012{
013    private RoomRandomizer aRandomRoom;
014    private ArrayList<Room> aRoomList;
015    private int aAleaInt;
016    
017    /**
018     * Constructeur qui permet d'initialiser les différents attributs
019     * @param pDescription la description de la pièce 
020     * @param pImageName le nom de l'image et son extension
021     * @param pRoomList la liste des pièces dans lesquels le joueur sera transporté
022     */
023    public TransporterRoom( final String pDescription, final String pImageName, final ArrayList pRoomList )
024    {
025        super(pDescription, pImageName);
026        this.aRoomList = pRoomList;
027        this.aRandomRoom = new RoomRandomizer(pRoomList);
028        this.aAleaInt = 17;
029    } // TransporterRoom(...)
030    
031    /**
032     * Accesseur pour récupérer une pièce aléatoirement
033     * @param pDirection la direction sous forme de chaine de caractère
034     * @return la pièce aléatoire qui sera utilisé pour transporter le joueur
035     */
036    @Override public Room getExit( final String pDirection )
037    {
038        if ( this.aAleaInt == 17 ) return this.findRandomRoom();
039        else return this.aRoomList.get(this.aAleaInt);
040    } // getExit(.)
041    
042    /**
043     * Modificateur qui permet de modifier la valeur de l'attribut aAleaInt par l'entier passé en paramètre
044     * @param pAleaInt l'entier qui modifiera l'attribut
045     */
046    public void setAleaInt( final int pAleaInt )
047    {
048       this.aAleaInt =  pAleaInt;
049    } // setAleaInt(.)
050
051    /**
052     * Procédure privée qui permet d'avoir une pièce prise de manière aléatoire
053     * @return un pièce de façon random
054     */
055    private Room findRandomRoom()
056    {
057        return this.aRandomRoom.findRandomRoom();
058    } // findRandomRoom()
059    
060} // TransporterRoom
