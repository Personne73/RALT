001package pkg_room;
002
003import java.util.Random;
004import java.util.ArrayList;
005/**
006 * Cette classe permet l'obtendtion un pièce de manière pseudo aléatoire
007 */
008public class RoomRandomizer
009{
010    private ArrayList<Room> aRoomList;
011    private Random aRandom;
012
013    /**
014     * Constructeur qui permet d'initialiser le générateur de nombre pseudo aléatoires
015     * ainsi que la liste de pièce
016     * @param pRoomList la liste des pièces
017     */
018    public RoomRandomizer( final ArrayList pRoomList )
019    {
020        this.aRoomList = pRoomList;
021        this.aRandom = new Random();
022    } // RoomRandomizer(.)
023
024    /**
025     * Méthode qui permet d'avoir une pièce prise de manière aléatoire dans la liste des pièces
026     * @return un pièce de façon random de la liste des pièces
027     */
028    public Room findRandomRoom()
029    {
030        int vRandomNumber = this.aRandom.nextInt(4);
031        return this.aRoomList.get(vRandomNumber);
032    } // findRandomRoom()
033
034} // RoomRandomizer
