001import pkg_game.GameEngine;
002import pkg_game.UserInterface;
003
004/**
005 * La classe Game permet d'initialiser le jeu en créant un moteur ainsi qu'une interface graphique
006 * 
007 * @author MBIAPA KETCHECKMEN Joël Trésor
008 * @version 2021.05
009 */
010public class Game
011{
012    private UserInterface aGui;
013    private GameEngine aEngine;
014   
015    /**
016     * Constructeur par défaut pour créer un nouveau jeu
017     */
018    public Game()
019    {
020        this.aEngine = new GameEngine();
021        this.aGui = new UserInterface(this.aEngine);
022        this.aEngine.setGUI(this.aGui);
023    } // Game()  
024    
025    /**
026     * Méthode principale du jeu
027     */
028    public static void main(String[] args){
029        new Game();
030    } // main(.)
031    
032} // Game
