002/**
003 * Cette classe permet la création des items du jeu
004 * 
005 * @author MBIAPA KETCHECKMEN Joël Trésor
006 * @version 2021.05
007 */
008public class pkg_item.Item
009{
010    private String aItemName;
011    private String aItemDescription;
012    private int aItemWeight; // en grammes pour l'instant
013    private int aItemPrice; // Monnaie d'échange : Coin 
014    
015    /**
016     * Constructeur naturel pour créer un nouvelle item
017     * @param pItemName le nom de l'item
018     * @param pItemDescription la description de l'item
019     * @param pItemWeight le poids de l'item
020     * @param pItemPrice le prix de l'item
021     */
022    public pkg_item.Item( final String pItemName, final String pItemDescription, 
023                final int pItemWeight, final int pItemPrice )
024    {
025        this.aItemName = pItemName;
026        this.aItemDescription = pItemDescription;
027        this.aItemWeight = pItemWeight;
028        this.aItemPrice = pItemPrice;
029    } // pkg_item.Item(....)
030
031    /**
032     * Accesseur pour accéder à la description de l'item
033     * @return la description de l'item
034     */
035    public String getItemDescription()
036    {
037        return this.aItemDescription;
038    } // getItemDesctiprion()
039    
040    /**
041     * Accesseur qui retourne une description complète de l'item
042     * @return la longue description de l'item
043     */
044    public String getLongItemDescription()
045    {
046        return "- " + this.getItemName() + "\n" + 
047        this.getItemDescription() + "\nPoids : " + 
048        this.getItemWeight() + " grammes  |  Prix : " 
049        + this.getItemPrice() + " Coins";
050    } // getLongItemDescription()
051
052    /**
053     * Accesseur pour accéder au nom de l'item
054     * @return le nom de l'item
055     */
056    public String getItemName()
057    {
058        return this.aItemName;
059    } // getItemName()
060    
061    /**
062     * Accesseur retourne le poids de l'item
063     * @return le poids de l'item
064     */
065    public int getItemWeight()
066    {
067        return this.aItemWeight;
068    } // getItemWeight()
069
070    /**
071     * Accesseur qui retourne le prix de l'item
072     * @return le prix de l'item
073     */
074    public int getItemPrice()
075    {
076        return this.aItemPrice;
077    } // getItemPrice()
078    
079} // pkg_item.Item
