134    /**
135     * Procédure qui change la salle courante en permettant
136     * au joueur de revenir dans une salle déjà visitée
137     */
138    public void back()
139    {
140        if ( this.aPreviousRooms.empty() ){
141            this.aGui.setEnableBackB(false);
142            this.aGui.println("Vous n'avez pas mémoriser de trajet, par conséquent, il vous est impossible " 
143                + "\nde retourner sur vos pas.");
144        } else {
145            if ( this.wasTransporterRoom() ){
146                this.aPreviousRooms.clear();
147                this.aGui.println("Vous venez d'une pièce transporteuse, vous n'avez pas mémorisez de trajet par conséquent !");
148                this.aGui.setEnableBackB(false);
149                return;
150            }
151            if ( this.aCurrentRoom.isExit(this.aPreviousRooms.peek()) ){
152                if ( this.getCurrentRoom() instanceof TransporterRoom ){
153                    TransporterRoom vRoom = (TransporterRoom)(this.getCurrentRoom());
154                    this.setCurrentRoom(vRoom.getExit("SudOuest"));
155                    this.aGui.println("Vous étiez dans une pièce transporteuse, sortir de cette pièce vous déplace aléatoirement !!");
156                    return;
157                }
158                this.aCurrentRoom = this.aPreviousRooms.pop();
159            } else {
160                this.aGui.println("Vous avez franchi une porte à sens unique. Il vous est impossible de retourner en arrière.");
161                this.aPreviousRooms.clear();
162                this.aGui.setEnableBackB(false);
163            }
164        }
165    } // back()

197    /**
198     * Procédure qui permet au joueur de prendre un item
199     * disponible dans la pièce courante
200     * @param pItemName le nom de l'item à prendre
201     */
202    public void take( final String pItemName )
203    {
204        Item vItem = this.aCurrentRoom.getItem(pItemName);
205        if ( vItem != null ) {
206            if ( vItem.getItemName().equals("Bracelet") ){
207                this.setMaxWeight( 150 );
208            }
209            if ( this.canBeTake(vItem) ){
210                this.addInventoryItem(vItem);
211                this.aCurrentRoom.removeItem(pItemName);
212                this.aGui.println("Vous avez pris l'item : " + pItemName);
213                this.aGui.println(this.inventoryList());
214                this.aGui.setEnableInventoryB(true);
215            } else if( this.aInventory.getItem("Bracelet") == null ){
216                this.aGui.println("Vous ne pouvez pas prendre d'item tant que vous ne possèdez pas le Bracelet de Stockage.");
217            } else {
218                this.aGui.println("L'indicateur de poids de votre Bracelet de Stockage s'agite :"
219                    + "\nvous avez dépassez la capacité disponible dans l'espace dimensionnel !!" 
220                    + "\nIl est impossible de prendre cet item au risque d'une saturation.");
221                this.aGui.setEnableInventoryB(true);
222            }
223        } else {
224            this.aGui.println("L'item que vous essayer " 
225                + "de prendre n'existe pas dans cette pièce !");
226        }
227    } // take(.)

253    /**
254     * Méthode booléen qui vérifie que l'item peut être pris par le joueur
255     * @param pItem item sur lequel la vérifiaction sera faite
256     * @return retourne false si le poids de l'item plus celui de l'inventaire est 
257     * supérieur au poids maximum supporter par le joueur, sinon retourne true
258     */
259    private boolean canBeTake( final Item pItem )
260    {
261        int vItemsWeight = this.aInventory.getItemListWeight() + pItem.getItemWeight();
262        if ( vItemsWeight > this.getMaxWeight() ) return false;
263        return true;
264    } // canBeTake()

274    /**
275     * Procédure qui vérifie que l'item associé au nom passer en paramètre peut
276     * être utiliser et réalise l'action que l'item es censé faire
277     * @param pItemName le nom de l'item que le joueur veut utiliser
278     */
279    public void use( final String pItemName )
280    {
281        Item vItem = this.getInventoryItem(pItemName);
282
283        if ( vItem != null ){
284            if ( pItemName.equals("Parchemin") ){
285                this.doubleMaxWeight();
286                this.removeInventoryItem(pItemName);
287                this.aGui.println("Le Parchemin entre en résonnance avec votre Bracelet de Stockage."
288                    + "\nUne lueur vous aveugle ! La capacité de votre Bracelet viens d'être doubler !!");
289                //this.aGui.println(this.inventory());
290            } else if ( pItemName.equals("Rune") ){
291                Beamer vBeamer = (Beamer)(vItem);
292                if ( this.getCurrentRoom().equals(vBeamer.getBeamerRoom()) ){
293                    this.aGui.println("Vous devez effectuez un déplacement avant de vous téléporter au risque de perdre définitivement la Rune");
294                    return;
295                }
296                this.fireBeamer(vBeamer);
297                this.aGui.println("Vous avez été téléporter dans la pièce mémorisé lors du chargement." 
298                    + "\nLa Rune de téléportation perd de son éclat, elle vient de se briser !!!");
299                //this.aGui.println(this.inventory());
300            } else if ( pItemName.equals("Talisman") ){
301                if ( vItem instanceof Beamer ){
302                    Beamer vBeamer = (Beamer)(vItem);
303                    this.fireBeamer(vBeamer);
304                    this.aGui.println("Vous avez été téléporter vers le lieu choisi par le Créateur."
305                        + "\nLe Talisman de téléportation s'efface de la réalité.");
306                }  else {
307                    this.aGui.println("Vous possedez une version incomplète du Talisman : le Plastron."
308                        + "\nRéunissez les pièces manquantes et assemblez le Talisman pour obtenir une version complète.");
309                }
310            }else {
311                this.aGui.println("Vous pouvez seulement utiliser un téléporteur ou un item d'augmentation de stockage");
312            }
313        } else {
314            this.aGui.println("L'item que vous essayer d'utiliser ne se trouve pas dans votre Bracelet.");
315        }
316    } // use(.)

329    /**
330     * Procédure qui permet de charger un téléporteur si le joueur est en possession de l'item
331     * passe en paramètre
332     * @param pBeamerName le nom de l'item à charger
333     */
334    public void chargeBeamer( final String pBeamerName )
335    {
336        Item vItem = this.getInventoryItem(pBeamerName);
337
338        if ( vItem != null ){ //le joueur possède l'item dans son inventaire
339            //le joueur possède un télépoteur
340            if ( vItem instanceof Beamer ){ // vérifie que vItem est une instance de la classe pkg_item.Beamer
341                Beamer vBeamer = (Beamer)(vItem);
342                if ( vBeamer.getBeamerRoom() == null ){
343                    vBeamer.setBeamerRoom(this.getCurrentRoom());
344                    this.aGui.println("Vous avez charger le téléporteur, il a mémoriser la pièce ou vous vous trouvez actuellement !");
345                } else {
346                    if ( vBeamer.getItemName().equals("Talisman") ){
347                        this.aGui.println("Le Talisman de téléportation, ne peut être chargé puisqu'il vous téléportera"
348                            + "\nvers le lieu prédéfini lors de sa création. Il a été charger par le Sorcier Suprême l'ayant fabriqué.");
349                        return;
350                    }
351                    this.aGui.println("L'item " + pBeamerName + " est déjà chargé !");
352                }
353            } else {
354                this.aGui.println("Vous pouvez seulement charger un téléporteur");
355            }
356        } else {
357            this.aGui.println("Vous ne possèdez pas l'item : " + pBeamerName);
358        }
359    } // chargeBeamer(.)

401    /**
402     * Fonction booleénne qui vérifie que le joueur possède les items nécéssaire pour assembler l'item qui 
403     * à pour nom celui passé en paramètre
404     * @param pItemName le nom de l'item à assembler
405     * @return vrai si l'item est assemblé, sinon retorne faux lorsque le joueur ne peux pas assembler l'item
406     */
407    public boolean build( final String pItemName ) 
408    {
409        Item vItem = this.getInventoryItem(pItemName);
410
411        if ( vItem != null ){
412            if ( vItem.getItemName().equals("Plastron") ){
413                if ( this.getInventoryItem("Fiole") != null && this.getInventoryItem("Vanadinite") != null ){
414                    this.removeInventoryItem("Fiole");
415                    this.removeInventoryItem("Vanadinite");
416                    this.removeInventoryItem("Plastron");
417                    this.aGui.println("Vous avez assemblé le Plastron, il est passé de l'état incomplet, à l'état complet.");
418                    this.aGui.println("Vous êtes désormais en possession d'un Talisman de téléportation complet.\nIl est prêt à être utilisé.");
419                    return true;
420                } else {
421                    this.aGui.println("Il vous manque la Fiole d'Énergie Lunaire ou la Vanadinite dans votre inventaire.");
422                    return false;
423                }
424            } else {
425                this.aGui.println("Vous pouvez seulement assemblez un Plastron de téléportation incomplet !");
426                return false;
427            }
428        } else {
429            this.aGui.println("Vous ne possédez pas l'item dans votre inventaire.");
430            return false;
431        }
432    } // build(.)

434    /**
435     * Procédure qui permet d'ouvrir la porte qui se trouve dans la direction passé en paramètre
436     * si il y a une porte
437     * @param pDirection la direction dans laquel on veut ouvrir une porte
438     */
439    public void openDoor( final String pDirection )
440    {
441        Door vDoor = this.getCurrentRoom().getDoor(pDirection); // la porte dans la direction souhaiter
442
443        if ( vDoor != null ){ // si il y a une porte dans la direction demandé
444            if ( !vDoor.getCanBeCrossed() ){ // si la porte ne peut pas être traversée
445                Item vKeyDoor = vDoor.getKeyDoor(); // la clé de la porte
446                Item vKeyInventory = this.getInventoryItem(vKeyDoor.getItemName()); // la clé de la porte dans l'inventaire 
447
448                if ( vKeyInventory != null ){ // le joueur possède la clé de la porte dans son inventaire
449                    vDoor.setCanBeCrossed(true);
450                    this.aGui.println("La porte dans la direction " + "\"" + pDirection + "\"" + " est maintenant dévérouillé.");
451                } else {
452                    this.aGui.println("Vous devez récupérer la clé " + "(" + vKeyDoor.getItemName() + ")" + " de la porte pour pouvoir l'ouvrir !");
453                }
454            } else {
455                this.aGui.println("Cette porte est déjà ouverte !");
456            }
457        } else {
458            this.aGui.println("Il n'y a pas de porte vérouillée dans cette direction.");
459        }
460    } // openDoor(.)
