059    /**
060     * Methode qui affiche du texte
061     * @param pText ligne de texte
062     */
063    public void print( final String pText )
064    {
065        this.aLog.append(pText);
066        this.aLog.setCaretPosition(this.aLog.getDocument().getLength());
067    } // print(.)
068
069    /**
070     * Méthode qui affiche le texte avec un saut de ligne
071     * @param pText ligne de texte
072     */
073    public void println( final String pText )
074    {
075        this.print(pText + "\n");
076    } // println(.)

078    /**
079     * Procédure qui permet de montrer une image dans l'interface
080     * @param pImageName le nom de l'image
081     */
082    public void showImage( final String pImageName )
083    {
084        String vImagePath = "Images/" + pImageName; // pour changer le repertoire
085        URL vImageURL = this.getClass().getClassLoader().getResource( vImagePath );
086        if ( vImageURL == null )
087            System.out.println( "Image non trouvée : " + vImagePath );
088        else {
089            ImageIcon vIcon = new ImageIcon( vImageURL );
090            this.aImage.setIcon( vIcon );
091            this.aMyFrame.pack();
092        }
093    } // showImage(.)
094
095    /**
096     * Active ou désactive la saisi dans le champ de saisi.
097     * @param pOnOff boolean qui active ou desactive la saisi
098     */
099    public void enable( final boolean pOnOff )
100    {
101        this.aEntryField.setEditable( pOnOff ); // enable/disable
102        if ( ! pOnOff ) { // disable
103            this.aEntryField.getCaret().setBlinkRate( 0 ); // cursor won't blink
104            this.aEntryField.removeActionListener( this ); // won't react to entry
105        }
106    } // enable(.)

108    /**
109     * Procédure qui crée l'interface graphique de l'utilisateur
110     */
111    private void createGUI()
112    {
113        this.aMyFrame = new JFrame( "R&LT - by 113 Block | Tokenizer" ); // change the title
114        this.aEntryField = new JTextField( 34 );
115
116        this.aLog = new JTextArea();
117        // permet de passer a la ligne lorsque l'on arrive au bout de l'écran
118        this.aLog.setLineWrap(true);// permet de couper la phrase lorsqu'on arrive a la limite de l'espace dispo pour faire un retour a la ligne
119        this.aLog.setWrapStyleWord(true);//permet de ne pas couper le mot lorsqu'on arrive a la fin de l'espace dispo et d'aller a la ligne
120
121        this.aLog.setEditable( false );
122        this.aListScroller = new JScrollPane( this.aLog );
123
124        JPanel vPanel = new JPanel();
125        this.aImage = new JLabel();
126        
127        //modification de la police
128        this.aLog.setFont(new Font("Calibri", Font.PLAIN, 12));
129        
130        //creation d'un panneau pour les boutons
131        JPanel vButtonPanel = new JPanel();
132
133        //création d'un panneau pour l'inventaire/affichage
134        JPanel vAffichagePanel = new JPanel();
135
136        //création d'une zone de texte pour l'inventaire
137        this.aInventory = new JTextArea();
138        this.aInventory.setEditable( false );
139        this.aInventory.setLineWrap(true);

141
142        //ajout de la zone de texte de l'inventaire dans le panneau scrollable 
143        this.aInventoryScroller = new JScrollPane( this.aInventory );

145        // création des boutons
146        JButton vHelpButton = new JButton("aide");
147        JButton vQuitButton = new JButton("quitter");

149        JButton vInventoryButton = new JButton("inventaire");
150        JButton vMap = new JButton("carte");

153        vHelpButton.setBackground( Color.darkGray );
154        vHelpButton.setForeground( Color.yellow );
155        vQuitButton.setBackground( Color.darkGray );

181        this.aButtonTab[3] = vQuitButton;

273    /**
274     * Fonction booléenne qui affiche la map si le booléen est mis a vrai sinon l'image de la pièce
275     * @param pVisibility le booléen indiquant la visibilité de la map
276     */
277    public void mapVisibility( final boolean pVisibility )
278    {
279        if ( pVisibility ){
280            for ( int i = 0; i<= 4; i++){
281                if( this.aEngine.getRoom(i).equals(this.aEngine.getPlayer().getCurrentRoom()) ) this.showImage("Etage1.png");
282            }
283            for ( int i = 5; i<= 10; i++){
284                if( this.aEngine.getRoom(i).equals(this.aEngine.getPlayer().getCurrentRoom()) ) this.showImage("Etage2.png");
285            }
286            for ( int i = 10; i<= 15; i++){
287                if( this.aEngine.getRoom(i).equals(this.aEngine.getPlayer().getCurrentRoom()) ) this.showImage("Etage3.png");
288            }
289        } else this.showImage(this.aEngine.getPlayer().getCurrentRoom().getImageName());
290    } // mapVisibility(.)

323    /**
324     * Porcédure qui vérifie le type de l'action passé en paramètre et réagis 
325     * en conséquence en fonction de s'il s'agit d'un clic ou d'un appuie 
326     * sur 'entré' dans le clavier
327     * @param pE l'action a vérifié
328     */
329    public void actionPerformed( final ActionEvent pE ) 
330    {
331        // no need to check the type of action at the moment
332        // because there is only one possible action (text input) :
333        for (int i=0; i<this.aButtonTab.length; i++) {
334            if ( pE.getSource() == this.aButtonTab[i] ){
335                this.aEngine.interpretCommand(  pE.getActionCommand() );
336                return;
337            } 
338        }
339        this.processCommand(); // never suppress this line
340    } // actionPerformed(.)
341
342    /**
343     * Procédure qui lis la commande et l'excécute 
344     */
345    private void processCommand()
346    {
347        String vInput = this.aEntryField.getText();
348        this.aEntryField.setText( "" );
349
350        this.aEngine.interpretCommand( vInput );
351    } // processCommand()
