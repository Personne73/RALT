001package pkg_item;

003 
004
005import pkg_room.Room;
006/**
007 * La classe Beamer permet la création d'items considérer comme téléporteur
008 * 
009 * @author MBIAPA KETCHECKMEN Joël Trésor
010 * @version 2021.05
011 */
012public class Beamer extends Item
013{
014    private Room aBeamerRoom;
015    
016    /**
017     * Constructeur qui initialise les attributs
018     * @param pItemName le nom de l'item
019     * @param pItemDescription la description de l'item
020     * @param pItemWeight le poids de l'item
021     * @param pItemPrice le prix de l'item
022     * @param pBeamerRoom la pièce à mémoriser par le beamer
023     */
024    public Beamer( final String pItemName, final String pItemDescription, 
025                    final int pItemWeight, final int pItemPrice, final Room pBeamerRoom )
026    {
027        super(pItemName, pItemDescription, pItemWeight, pItemPrice);
028        this.aBeamerRoom = pBeamerRoom;
029    } // Beamer(.....)
030    
031    /**
032     * Constructeur qui initialise les attributs de la super classe
033     * ainsi que celui de la sous classe en le mettant à null
034     * @param pItemName le nom de l'item
035     * @param pItemDescription la description de l'item
036     * @param pItemWeight le poids de l'item
037     * @param pItemPrice le prix de l'item
038     */
039    public Beamer( final String pItemName, final String pItemDescription, 
040                    final int pItemWeight, final int pItemPrice )
041    {
042        this(pItemName, pItemDescription, pItemWeight, pItemPrice, null);
043    } // Beamer(....)
044    
045    /**
046     * Accesseur à la valeur qui se trouve dans l'attribut aBeamerRoom
047     * @return la valeur stocker dans l'attribut aBeamerRoom
048     */
049    public Room getBeamerRoom()
050    {
051        return this.aBeamerRoom;
052    } // getBeamerRoom()
053    
054    /**
055     * Modificateur qui modifie la valeur de l'attribut aBeamerRoom
056     * @param pBeamerRoom la pièce utilisé pour modifier la valeur de l'attribut
057     */
058    public void setBeamerRoom( final Room pBeamerRoom )
059    {
060        this.aBeamerRoom = pBeamerRoom;
061    } // setBeamerRoom(.)
062    
063} // Beamer
