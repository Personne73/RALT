003/**
004 * La classe pkg_room.pkg_room.Door permet de créer des portes dans notre jeu
005 * 
006 * @author MBIAPA KETCHECKMEN Joël Trésor
007 * @version 2021.05
008 */
009public class pkg_room.pkg_room.Door
010{
011    private boolean aCanBeCrossed;
012    private pkg_item.Item aKeyDoor;
013    
014    /**
015     * Constructeur naturel qui permet d'initialiser ses attributs
016     * @param pCanBeCrossed vaut vrai si la porte peut être franchi, sinon faux
017     * @param pKeyDoor représente la clé de la porte
018     */
019    public pkg_room.pkg_room.Door( final boolean pCanBeCrossed, final pkg_item.Item pKeyDoor )
020    {
021        this.aCanBeCrossed = pCanBeCrossed;
022        this.aKeyDoor = pKeyDoor;
023    } // pkg_room.pkg_room.Door(..)
024    
025    /**
026     * Accesseur qui permet d'avoir la valeur contenu dans l'attribut 
027     * aCanBeCrossed
028     * @return la valeur de l'attribut aCanBeCrossed
029     */
030    public boolean getCanBeCrossed()
031    {
032        return this.aCanBeCrossed;
033    } // getCanBeCrossed()
034    
035    /**
036     * Accesseur qui permet de savoir si la porte possède une clé ou non
037     * @return la valeur de l'attribut aCanBeCrossed
038     */
039    public pkg_item.Item getKeyDoor()
040    {
041        return this.aKeyDoor;
042    }  // getKeyDoor()
043    
044    /**
045     * Modificateur qui permet de changer la valeur contenue dans l'attribut
046     * aCanBeCrossed
047     * @param pCanBeCrossed la nouvelle valeur de l'attribut canBeCrossed
048     */
049    public void setCanBeCrossed( final boolean pCanBeCrossed )
050    {
051        this.aCanBeCrossed = pCanBeCrossed;
052    } // setCanBeCrossed(.)
053} // pkg_room.pkg_room.Door
