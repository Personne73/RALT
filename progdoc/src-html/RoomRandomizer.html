004/**
005 * Cette classe permet l'obtendtion un pièce de manière pseudo aléatoire

007public class pkg_room.pkg_room.pkg_room.RoomRandomizer
008{
009    private ArrayList<pkg_room.pkg_room.pkg_room.Room> aRoomList;
010    private Random aRandom;
011    
012    /**
