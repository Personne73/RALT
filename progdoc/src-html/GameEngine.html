204    /**
205     * Procédure pour changer de pièce à partir du mot de commande
206     * @param pCommand le mot de commande à traiter
207     */
208    private void goRoom( final pkg_command.Command pCommand )
209    {
210        if ( !pCommand.hasSecondWord() ) {
211            this.aGui.println("Aller où ?");
212            return;
213        } 
214
215        String vDirection = pCommand.getSecondWord();
216
217        pkg_room.pkg_room.pkg_room.Room vNextRoom = this.aPlayer.getCurrentRoom().getExit(vDirection);
218
219        if (vNextRoom == null) {
220            this.aGui.println("Il n'y a pas de porte !");
221            return;
222        }
223
224        // si il y a une salle dans la direction, on test maintenant si il y a une porte
225        pkg_room.pkg_room.Door vNextRoomDoor = this.aPlayer.getCurrentRoom().getDoor(vDirection);
226        if ( vNextRoomDoor != null ){
227            if ( vNextRoomDoor.getCanBeCrossed() == false ){ // si la porte  ne peut pas être traversé
228                this.aGui.println("La porte est vérouillé, vous devez trouver la clé et la dévérouiller pour pouvoir passer !");
229                return;
230            } else {
231                this.aPlayer.setCurrentRoom(vNextRoom);
232                vNextRoomDoor.setCanBeCrossed(false);
233            }
234        }
235
236        this.aPlayer.setCurrentRoom(vNextRoom);
237        this.printLocationInfo();
238    } // goRoom(.)

273    /**
274     * Procédure qui exécute la méthode appropriée à la commande rentrée
275     * @param pCommandLine la ligne de commande
276     */
277    public void interpretCommand( final String pCommandLine ) 
278    {
279        this.aGui.println("\n> " + pCommandLine);
280        pkg_command.Command vCommand = this.aParser.getCommand( pCommandLine );
281
282        if ( vCommand.isUnknown() ){
283            this.aGui.println("Je ne sais pas de quoi vous parlez...");
284            return;
285        } 
286
287        if ( vCommand.getCommandWord().equals("aller") ){
288            this.aGui.visibilityInventory(false);
289            this.goRoom(vCommand);          
290        } else if ( vCommand.getCommandWord().equals("quitter") ){
291
292            if ( vCommand.hasSecondWord() ){
293                this.aGui.println("Quitter quoi ?");
294                return;
295            } else {
296                this.aGui.visibilityInventory(false);
297                this.endGame();
298            }
299
300        } else if ( vCommand.getCommandWord().equals("aide") ){
301            this.aGui.visibilityInventory(false);
302            this.printHelp();
303        } else if ( vCommand.getCommandWord().equals("regarder") ){
304            this.aGui.visibilityInventory(false);
305            this.look(vCommand);
306        } else if ( vCommand.getCommandWord().equals("manger") ){
307            this.aGui.visibilityInventory(false);
308            this.eat();
309        } else if ( vCommand.getCommandWord().equals("retour") ){
310            this.aGui.visibilityInventory(false);
311            this.back(vCommand);
312        } else if ( vCommand.getCommandWord().equals("test") ){
313            this.aGui.visibilityInventory(false);
314            this.test(vCommand);
315        } else if ( vCommand.getCommandWord().equals("prendre") ){
316            this.take(vCommand);
317            this.aGui.visibilityInventory(false);
318        } else if ( vCommand.getCommandWord().equals("deposer") ){
319            this.aPlayer.drop(vCommand.getSecondWord());
320        } else if ( vCommand.getCommandWord().equals("inventaire") ){
321            this.inventory();
322        } else if ( vCommand.getCommandWord().equals("utiliser") ){
323            this.aGui.visibilityInventory(false);
324            this.use(vCommand);
325        } else if ( vCommand.getCommandWord().equals("inspecter") ){
326            //this.aGui.visibilityInventory(false);
327            this.inspectItem(vCommand);
328        } else if ( vCommand.getCommandWord().equals("charger") ){
329            this.aGui.visibilityInventory(false);
330            this.chargeBeamer(vCommand);
331        } else if ( vCommand.getCommandWord().equals("assembler") ){
332            this.aGui.visibilityInventory(false);
333            this.build(vCommand);
334        } else if ( vCommand.getCommandWord().equals("alea") ){
335            this.aGui.visibilityInventory(false);
336            this.alea(vCommand);
337        }else if ( vCommand.getCommandWord().equals("ouvrir") ){
338            this.aGui.visibilityInventory(false);
339            this.openDoor(vCommand);
340        }
341    } // interpretCommand(.)

393    /**
394     * Procédure qui vérifie qu'un second mot de command est présent
395     * pour accéder au fichier ayant pour nom le second mot de commande
396     * de manière à le Scanner et lire les instructions qui s'y trouvent
397     * @param pCommand la ligne de commande
398     */
399    private void test( final pkg_command.Command pCommand )
400    {
401        //création d'une variable qui représente le fichier de texte
402        if ( pCommand.hasSecondWord() ){
403            this.aTestMode = true;
404            Scanner vFileText;
405            String vFileName = pCommand.getSecondWord();
406
407            //Association du fichier a la variable vFileText 
408            //try : pour essayer les instructions
409            try { 
410                vFileText = new Scanner( new File(vFileName + ".txt") ); 
411                //while : pour traiter la ligne lue
412                while ( vFileText.hasNextLine() ){
413                    String vLine = vFileText.nextLine();
414                    this.aGui.println(vLine);
415                    this.interpretCommand( vLine );
416                }
417            } 
418            catch ( final FileNotFoundException pFNFE ){
419                System.out.println(pFNFE);
420            }  
421            this.aTestMode = false;
422        } else {
423            this.aGui.println("Je ne sais que tester des fichiers .txt");
424        }
425    } // test(.)
