001
002import java.util.HashMap;
003import java.util.Set;
004/**
005 * Cette classe permet de créer une liste associant un nom et son item
006 * 
007 * @author MBIAPA KETCHECKMEN Joël Trésor
008 * @version 2021.05
009 */
010public class pkg_item.ItemList
011{
012    private HashMap<String, pkg_item.Item> aItemList;
013
014    /**
015     * Constructeur qui initialise l'attribut
016     */
017    public pkg_item.ItemList()
018    {
019        this.aItemList = new HashMap<String, pkg_item.Item>();
020    } // pkg_item.ItemList()
021
022    /**
023     * Accesseur qui retourne l'item de la liste correspondant
024     * à celui rentré en paramètre
025     * @param pItemName le nom de l'item recherché
026     * @return l'item possédant le nom passé en paramètre
027     */
028    public pkg_item.Item getItem( final String pItemName ) 
029    {
030        return this.aItemList.get(pItemName);
031    } // getItem(.)
032
033    /**
034     * Accesseur qui retourne la liste des items contenus dans liste
035     * accompagnés de leur poids
036     * @return une chaine de caractère contenant les items et leur poids
037     */
038    public String getItemList()
039    {
040        if( this.aItemList.size() > 0 ){
041            String vInventory = "Vous possédez :\n";
042            Set<String> vKeys = this.aItemList.keySet();
043            for(String vItemName : vKeys){
044                pkg_item.Item vReturn = this.aItemList.get(vItemName);
045                vInventory += "- " + vReturn.getItemName() + " (Poids : " + vReturn.getItemWeight() + " grammes)\n";
046            }
047            return vInventory + "Poids de tout les items stockés : " + this.getItemListWeight() + " grammes\n";
048        }
049        return "L'inventaire est vide.";
050    } // getItemList()
051
052    /**
053     * Accesseur qui retourne une chaine de caractère
054     * contenant tout les items de la liste accompagnés 
055     * de leurs descriptions
056     * @return les items de la liste et leurs descriptions.
057     */
058    public String getItemListString()
059    {
060        if( this.aItemList.size() > 0){
061            String vItemsDescriptions = "pkg_item.Item(s) disponible(s) :\n";
062            Set<String> vKeys = this.aItemList.keySet();
063
064            for(String vItemName : vKeys){
065                pkg_item.Item vReturn = this.aItemList.get(vItemName);
066                vItemsDescriptions += vReturn.getLongItemDescription() + "\n";
067            }
068            return vItemsDescriptions;
069        }
070        return "Il n'y a pas d'item(s)";
071    } // getItemListString()
072
073    /**
074     * Accesseur qui retourne la somme des poids de tout les items contenus
075     * dans la liste
076     * @return le poids de la liste d'item
077     */
078    public int getItemListWeight()
079    {
080        int vItemsWeight = 0;
081        Set<String> vKeys = this.aItemList.keySet();
082        for(String vItemName : vKeys){
083            pkg_item.Item vReturn = this.aItemList.get(vItemName);
084            vItemsWeight += vReturn.getItemWeight();
085        } 
086        return vItemsWeight;
087    } // getItemListWeight()
088
089    /**
090     * Procédure qui ajoute des items à la liste d'items
091     * @param pItem l'item à ajouter
092     */
093    public void addItemList( final pkg_item.Item pItem )
094    {
095        this.aItemList.put(pItem.getItemName(), pItem);
096    } // addItem(.)
097
098    /**
099     * Procédure qui retire un item de la liste des items
100     * @param pItemName le nom de l'item à enlever
101     */
102    public void removeItemList( final String pItemName )
103    {
104        this.aItemList.remove(pItemName);
105    } // removeItemList(.)
106
107} // pkg_item.ItemList
