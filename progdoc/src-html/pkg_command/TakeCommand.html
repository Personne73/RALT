003 
004
005import pkg_game.GameEngine;
006
007/**
008 * Cette classe représente la commande prendre
009 *
010 * @author MBIAPA KETCHECKMEN Joël Trésor
011 * @version 2021.05
012 */
013public class TakeCommand extends Command
014{
015
016    /**
017     * Constructeur pour créer une commande prendre
018     */
019    public TakeCommand()
020    {
021       
022    } // TakeCommand()
023
024    /**
025     * Fonction booléenne qui vérifie que la command possède un second mot
026     * avant de prendre l'objet
027     * @param pEngine le moteur du jeu pour accèder aux éléménts nécéssaires
028     * @return toujours faux
029     */
030    @Override public boolean execute( final GameEngine pEngine)
031    {
032        pEngine.getGui().visibilityInventory(false);
033        pEngine.getGui().mapVisibility(false);
034        if ( this.hasSecondWord() ){
035            pEngine.getPlayer().take( this.getSecondWord() );
036            if ( pEngine.getPlayer().getInventoryItem("Bague") != null ){
037                pEngine.getGui().println("Vous avez rempli votre mission. Vous rentrez à présent livrer la relique à vos employeurs américains.");
038                pEngine.endGame();
039                return true;
040            }
041        } else {
042            pEngine.getGui().println("Vous devez spécifier l'item " +
043                "que vous voulez prendre.");
044        }
045        return false;
046    } // execute(.)
047} // TakeCommand
