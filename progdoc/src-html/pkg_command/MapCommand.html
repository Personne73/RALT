001package pkg_command;
002
003import pkg_game.GameEngine;
004/**
005 * Cette classe représente la commande carte.
006 *
007 * @author MBIAPA KETCHECKMEN Joël Trésor
008 * @version 2021.05
009 */
010public class MapCommand extends Command
011{
012
013    /**
014     * Constructeur pour créer une commande carte
015     */
016    public MapCommand()
017    {
018    } // MapCommand()
019
020    /**
021     * Fonction booléenne qui vérifie que la ligne de commande passé en paramètre ne possède
022     * pas de second mot pour pouvoir assembler l'item ayant pour nom ce second mot
023     * @param pEngine le moteur du jeu pour accèder aux éléménts nécéssaires
024     * @return toujours faux
025     */
026    @Override public boolean execute( final GameEngine pEngine )
027    {
028        if ( this.hasSecondWord() ){
029            pEngine.getGui().println("La commande 'carte' ne prend pas de second mot !");
030        } else {
031            pEngine.getGui().mapVisibility(true);
032        }
033        return false;
034    } // execute(.)
035} // MapCommand
