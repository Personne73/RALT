005import pkg_game.GameEngine;
006import pkg_item.Item;
007
008/**
009 * Cette classe représente la commande regarder.
010 *
011 * @author MBIAPA KETCHECKMEN Joël Trésor
012 * @version 2021.05
013 */
014public class LookCommand extends Command
015{
016    
017
018    /**
019     * Constructeur pour créer la commande regarder
020     */
021    public LookCommand()
022    {
023      
024    } // LookCommand()
025
026    /**
027     * Fonction booléenne qui vérifie le second mot lorsqu'il existe pour regarder un item,
028     * sinon pour regarder ou le joueur se trouve actuellement
029     * @param pEngine le moteur du jeu pour accèder aux éléménts nécéssaires
030     * @return toujours faux
031     */
032    @Override public boolean execute( final GameEngine pEngine)
033    {
034        pEngine.getGui().visibilityInventory(false);
035        pEngine.getGui().mapVisibility(false);
036        if ( this.hasSecondWord() ){
037            Item vItem = pEngine.getPlayer().getCurrentRoom().getItem(this.getSecondWord());
038            if ( vItem == null ){
039                pEngine.getGui().println("Je ne sais regarder qu'un item existant !");
040            } else{
041                pEngine.getGui().println( vItem.getLongItemDescription() );
042            }
043        } else {
044            pEngine.printLocationInfo();
045        }
046        return false;
047    } // execute(.)
048} // LookCommand
