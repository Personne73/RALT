005import java.io.File;
006import java.io.FileNotFoundException;
007import java.util.Scanner;
008import pkg_game.GameEngine;
009/**
010 * Cette classe représente la commande test
011 *
012 * @author MBIAPA KETCHECKMEN Joël Trésor
013 * @version 2021.05
014 */
015public class TestCommand extends Command
016{
017    
018
019    /**
020     * Constructeur pour créer une commande test
021     */
022    public TestCommand()
023    {
024        
025    } // TestCommand()
026
027    /**
028     * Fonction booléenne qui vérifie qu'un second mot de command est présent
029     * pour accéder au fichier ayant pour nom ce second mot de commande
030     * de manière à le Scanner et lire les instructions qui s'y trouvent
031     * @param pEngine le moteur du jeu pour accèder aux éléménts nécéssaires
032     * @return toujours faux
033     */
034    @Override public boolean execute( final GameEngine pEngine)
035    {
036        pEngine.getGui().visibilityInventory(false);
037        if ( this.hasSecondWord() ){
038            pEngine.setTestMode(true);
039            Scanner vFileText;
040            String vFileName = this.getSecondWord();
041
042            //Association du fichier a la variable vFileText 
043            //try : pour essayer les instructions
044            try { 
045                vFileText = new Scanner( new File(vFileName + ".txt") ); 
046                //while : pour traiter la ligne lue
047                while ( vFileText.hasNextLine() ){
048                    String vLine = vFileText.nextLine();
049                    pEngine.getGui().println(vLine);
050                    pEngine.interpretCommand( vLine );
051                }
052            } 
053            catch ( final FileNotFoundException pFNFE ){
054                System.out.println(pFNFE);
055            }  
056            pEngine.setTestMode(false);
057        } else {
058            pEngine.getGui().println("Je ne sais que tester des fichiers .txt");
059        }
060        return false;
061    } // execute(.)
062} // TestCommand
