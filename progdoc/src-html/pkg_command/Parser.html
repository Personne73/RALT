001package pkg_command;

003 
004import pkg_command.Command;
005import pkg_command.CommandWords;
006import java.util.StringTokenizer;
007/**
008 * La classe Parser permet de créer les lignes de commmandes contenant ce que
009 * l'utilisateur rentre pour permettre leur lecture et interprétation
010 * 
011 * @author MBIAPA KETCHECKMEN Joël Trésor
012 * @version 2021.05
013 */
014public class Parser 
015{
016    private CommandWords aValidCommands;
017
018    /**
019     * Constructeur par defaut qui intialise l'attribut de type 
020     * CommandWords pour lui mettre les mots de commandes
021     */
022    public Parser() 
023    {
024        this.aValidCommands = new CommandWords();        
025    } // Parser()
026
027    /**
028     * Prend la prochaine commande rentrée par l'utilisateur
029     * @param pInputLine la lgine de commande
030     * @return la prochaine commande.
031     */
032    public Command getCommand( final String pInputLine )
033    {
034        // pInputLine contiendra toute la ligne tapée
035        String vWord1;
036        String vWord2;
037
038        // cherche jusqu'à 2 mots dans la ligne tapee
039        StringTokenizer vTokenizer = new StringTokenizer( pInputLine );
040        if ( vTokenizer.hasMoreTokens() ) {
041            vWord1 = vTokenizer.nextToken(); // recupere le premier mot
042        } else vWord1 = null;
043
044        if ( vTokenizer.hasMoreTokens() ) {
045            vWord2 = vTokenizer.nextToken();  // recupere le deuxieme mot
046            // note : on ignore tout le reste de la ligne tapee !
047        } else vWord2 = null;
048
049        /* Verifie si le premier mot est une commande connue. 
050        Si oui, cree une Command avec.
051        Sinon, cree une commande vide avec "null" 
052        (pour dire 'commande inconnue'). */
053        
054        Command vCommand = this.aValidCommands.getCommand(vWord1);
055        if ( vCommand != null) {
056            vCommand.setSecondWord(vWord2);
057        }
058        return vCommand;
059    } // getCommand(.)
060
061    /**
062     * Méthode qui retourne une liste des commandes valides
063     * @return les commandes valides en String
064     */
065    public String getCommands() //anciennement showCommands()
066    {
067        return this.aValidCommands.getCommandList();
068    } // getCommands() 
069} // Parser
