003 
004
005import pkg_game.GameEngine;
006
007/**
008 * Cette classe représente la commande ouvrir
009 *
010 * @author MBIAPA KETCHECKMEN Joël Trésor
011 * @version 2021.05
012 */
013public class OpenCommand extends Command
014{
015
016    /**
017     * Constructeur pour créer une commande ouvrir
018     */
019    public OpenCommand()
020    {
021        
022    } // OpenCommand()
023
024    /**
025     * Fonction booléenne qui vérifie que la ligne de commande passé en paramètre possède une second mot
026     * pour pouvoir ouvrir la porte dans la direction contenu dans ce second mot
027     * @param pEngine le moteur du jeu pour accèder aux éléménts nécéssaires
028     * @return toujours faux
029     */
030    @Override public boolean execute( final GameEngine pEngine )
031    {
032        pEngine.getGui().visibilityInventory(false);
033        pEngine.getGui().mapVisibility(false);
034        if ( this.hasSecondWord() ){
035            pEngine.getPlayer().openDoor(this.getSecondWord());
036        } else {
037            pEngine.getGui().println("La porte de quelle direction voulez-vous ouvrir ?");
038        }
039        return false;
040    } // execute(.)
041} // OpenCommand
