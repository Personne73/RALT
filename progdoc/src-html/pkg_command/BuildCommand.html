003

005import pkg_game.GameEngine;
006
007/**
008 * Cette classe représente la commande assembler.
009 *
010 * @author MBIAPA KETCHECKMEN Joël Trésor
011 * @version 2021.05
012 */
013public class BuildCommand extends Command
014{
015    
016
017    /**
018     * Constructeur pour créer une commande assembler
019     */
020    public BuildCommand()
021    {
022        
023    } // BuildCommand()
024
025    /**
026     * Fonction booléenne qui vérifie que la ligne de commande passé en paramètre possède
027     * un second mot pour pouvoir assembler l'item ayant pour nom ce second mot
028     * @param pEngine le moteur du jeu pour accèder aux éléménts nécéssaires
029     * @return toujours faux
030     */
031    @Override public boolean execute( final GameEngine pEngine )
032    {
033        pEngine.getGui().visibilityInventory(false);
034        pEngine.getGui().mapVisibility(false);
035        if ( this.hasSecondWord() ){
036            if ( pEngine.getPlayer().build(this.getSecondWord()) ){
037                pEngine.getPlayer().addInventoryItem(pEngine.getTalisman());
038                pEngine.getGui().println(pEngine.getPlayer().inventoryList());
039            }
040        } else {
041            pEngine.getGui().println("Quel item voulez-vous à assembler ?");
042        }
043        return false;
044    } // execute(.)
045} // BuildCommand
