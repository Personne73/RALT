001package pkg_command;

003 
004
005import pkg_game.GameEngine;
006
007/**
008 * Cette classe représente la command inventaire.
009 *
010 * @author MBIAPA KETCHECKMEN Joël Trésor
011 * @version 2021.05
012 */
013public class InventoryCommand extends Command
014{
015   
016
017    /**
018     * Constructeur pour créer une commande inventaire
019     */
020    public InventoryCommand()
021    {
022        
023    } // InventoryCommand()
024
025    /**
026     * Fonction booléenne qui affiche l'inventaire du joueur en appellant
027     * @param pEngine le moteur du jeu pour accèder aux éléménts nécéssaires
028     * @return toujours faux
029     */
030    @Override public boolean execute( final GameEngine pEngine)
031    {
032        pEngine.getGui().mapVisibility(false);
033        if ( this.hasSecondWord() ){
034            pEngine.getGui().println("La commande inventaire n'accepte pas de second mot.");
035            return false;
036        } 
037        pEngine.getPlayer().inventory();
038        return false;
039    } // execute(.)
040} // InventoryCommand
