001package pkg_command;

003 
004
005import java.util.HashMap;
006import java.util.Set;
007/**
008 * La classe CommandWords permet d'ajouter des mots de commandes aux commandes valides acceptées dans le jeu
009 * 
010 * @author MBIAPA KETCHECKMEN Joël Trésor
011 * @version 2021.05
012 */
013public class CommandWords
014{
015    private HashMap<String, Command> aValidCommands;
016
017    /**
018     * Constructeur par défaut qui initialise les mots de commandes
019     */
020    public CommandWords()
021    {
022        this.aValidCommands = new HashMap<String, Command>();
023        this.aValidCommands.put("aller", new GoCommand());
024        this.aValidCommands.put("aide", new HelpCommand(this));
025        this.aValidCommands.put("quitter", new QuitCommand());
026        this.aValidCommands.put("regarder", new LookCommand());
027        this.aValidCommands.put("manger", new EatCommand());
028        this.aValidCommands.put("retour", new BackCommand());
029        this.aValidCommands.put("test", new TestCommand());
030        this.aValidCommands.put("prendre", new TakeCommand());
031        this.aValidCommands.put("deposer", new DropCommand());
032        this.aValidCommands.put("inventaire", new InventoryCommand());
033        this.aValidCommands.put("inspecter", new InspectCommand());
034        this.aValidCommands.put("utiliser", new UseCommand());
035        this.aValidCommands.put("charger", new ChargeCommand());
036        this.aValidCommands.put("assembler", new BuildCommand());
037        this.aValidCommands.put("alea", new AleaCommand());
038        this.aValidCommands.put("ouvrir", new OpenCommand());
039        this.aValidCommands.put("carte", new MapCommand());
040    } // CommandWords()
041
042    /**
043     * Accesseur qui permet de récupérer la commande qui possède pour nom celui
044     * passé en paramètre
045     * @param pCommandWord le nom de la commande à récupérer
046     * @return la commande associé au nom rentré en paramètre
047     */
048    public Command getCommand( final String pCommandWord )
049    {
050        return this.aValidCommands.get(pCommandWord);
051    } // getCommand(.)
052
053    /**
054     * Méthode qui renvoie une chaine de carractère 
055     * des commandes disponibles
056     * @return les commandes sous forme de chaine de carectère
057     */
058    public String getCommandList() // fut showAll pendant un moment
059    {
060        String vCommandString = "";
061        Set<String> vCommandWords = this.aValidCommands.keySet();
062        for (String vCommandWord : vCommandWords ){
063            vCommandString += vCommandWord + " ";
064        }
065        return vCommandString;
066    } // getCommandList()
067} // CommandWords
