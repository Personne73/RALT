005import pkg_game.GameEngine;
006
007/**
008 * Cette classe représente la commande d'aide
009 *
010 * @author MBIAPA KETCHECKMEN Joël Trésor
011 * @version 2021.05
012 */
013public class HelpCommand extends Command
014{
015    private CommandWords aCommandWords;
016
017    /**
018     * Constructeur pour créer une commande aide
019     * @param pWords les mots de commandes à afficher
020     */
021    public HelpCommand( final CommandWords pWords )
022    {
023        this.aCommandWords = pWords;
024    } // HelpCommand(.)
025
026    /**
027     * Focntion booléenne qui affiche les commandes disponibles lorsque le joueur demande de l'aide
028     * @param pEngine le moteur du jeu pour accèder aux éléménts nécéssaires
029     * @return toujours faux
030     */
031    @Override public boolean execute( final GameEngine pEngine )
032    {
033        pEngine.getGui().visibilityInventory(false);
034        pEngine.getGui().mapVisibility(false);
035        pEngine.getGui().println("Vous êtes perdu. Vous êtes seul.");
036        pEngine.getGui().println("Vous errez dans la Tour Perdu.");
037        pEngine.getGui().println("\n\nVos mots de commandes sont:\n");
038        pEngine.getGui().println(pEngine.getParser().getCommands() );
039        return false;
040    } // execute(.)
041} // HelpCommand
