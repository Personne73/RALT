001package pkg_command;

003 
004
005import pkg_game.GameEngine;
006import pkg_room.Room;
007import pkg_room.TransporterRoom;
008
009/**
010 * La classe AleaCommand représente la commande 'alea'
011 *
012 * @author MBIAPA KETCHECKMEN Joël Trésor
013 * @version 2021.05
014 */
015public class AleaCommand extends Command
016{
017
018    /**
019     * Constructeur pour créer une commande alea
020     */
021    public AleaCommand()
022    {
023        
024    } // AleaCommand()
025
026    /**
027     * Fonction booléenne qui vérifie que la ligne de commande possède un second mot pour pouvoir
028     * modifier le caractère aléatoire de la tranporter room, ou lui rendre ce caractère si elle
029     * ne possède pas de second mot
030     * @param pEngine le moteur du jeu pour accèder aux éléménts nécéssaires
031     * @return toujours faux
032     */
033    @Override public boolean execute( final GameEngine pEngine )
034    {
035        pEngine.getGui().visibilityInventory(false);
036        Room vRoom = pEngine.getRoom(4);
037        TransporterRoom vNextRoom = (TransporterRoom)(vRoom);
038        if ( this.hasSecondWord() ){
039            if ( pEngine.getTestMode() ){
040                int vInt = Integer.valueOf(this.getSecondWord());
041                vNextRoom.setAleaInt(vInt);
042            } else pEngine.getGui().println("Cette commande ne peut être utilisée qu'en mode test.");
043        } else {
044            if ( pEngine.getTestMode() ){
045                vNextRoom.setAleaInt(17);
046            } else pEngine.getGui().println("Cette commande ne peut être utilisée qu'en mode test.");
047        } 
048        return false;
049    } // execute(.)
050} // AleaCommand
