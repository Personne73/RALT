128    /**
129     * Procédure qui change la salle courante en permettant
130     * au joueur de revenir dans une salle déjà visitée
131     */
132    public void back()
133    {
134        if ( this.aPreviousRooms.empty() ){
135            this.aGui.setEnableBackB(false);
136            this.aGui.println("Vous n'avez pas mémoriser de trajet, par conséquent, il vous est impossible " 
137                + "\nde retourner sur vos pas.");
138        } else {
139            if ( this.wasTransporterRoom() ){
140                this.aPreviousRooms.clear();
141                this.aGui.println("Vous venez d'une pièce transporteuse, vous n'avez pas mémorisez de trajet par conséquent !");
142                this.aGui.setEnableBackB(false);
143                return;
144            }
145            if ( this.aCurrentRoom.isExit(this.aPreviousRooms.peek()) ){
146                if ( this.getCurrentRoom() instanceof pkg_room.pkg_room.pkg_room.TransporterRoom ){
147                    pkg_room.pkg_room.pkg_room.TransporterRoom vRoom = (pkg_room.pkg_room.pkg_room.TransporterRoom)(this.getCurrentRoom());
148                    this.setCurrentRoom(vRoom.getExit("SudOuest"));
149                    this.aGui.println("Vous étiez dans une pièce transporteuse, sortir de cette pièce vous déplace aléatoirement !!");
150                    return;
151                }
152                this.aCurrentRoom = this.aPreviousRooms.pop();
153            } else {
154                this.aGui.println("Vous avez franchi une porte à sens unique. Il vous est impossible de retourner en arrière.");
155                this.aPreviousRooms.clear();
156                this.aGui.setEnableBackB(false);
157            }
158        }
159    } // back()

191    /**
192     * Procédure qui permet au joueur de prendre un item
193     * disponible dans la pièce courante
194     * @param pItemName le nom de l'item à prendre
195     */
196    public void take( final String pItemName )
197    {
198        pkg_item.Item vItem = this.aCurrentRoom.getItem(pItemName);
199        if ( vItem != null ) {
200            if ( vItem.getItemName().equals("Bracelet") ){
201                this.setMaxWeight( 150 );
202            }
203            if ( this.canBeTake(vItem) ){
204                this.addInventoryItem(vItem);
205                this.aCurrentRoom.removeItem(pItemName);
206                this.aGui.println("Vous avez pris l'item : " + pItemName);
207                this.aGui.println(this.inventory());
208                this.aGui.setEnableInventoryB(true);
209            } else if( this.aInventory.getItem("Bracelet") == null ){
210                this.aGui.println("Vous ne pouvez pas prendre d'item tant que vous ne possèdez pas le Bracelet de Stockage.");
211            } else {
212                this.aGui.println("L'indicateur de poids de votre Bracelet de Stockage s'agite :"
213                    + "\nvous avez dépassez la capacité disponible dans l'espace dimensionnel !!" 
214                    + "\nIl est impossible de prendre cet item au risque d'une saturation.");
215                this.aGui.setEnableInventoryB(true);
216            }
217        } else {
218            this.aGui.println("L'item que vous essayer " 
219                + "de prendre n'existe pas dans cette pièce !");
220        }
221    } // take(.)

247    /**
248     * Méthode booléen qui vérifie que l'item peut être pris par le joueur
249     * @param pItem item sur lequel la vérifiaction sera faite
250     * @return retourne false si le poids de l'item plus celui de l'inventaire est 
251     * supérieur au poids maximum supporter par le joueur, sinon retourne true
252     */
253    private boolean canBeTake( final pkg_item.Item pItem )
254    {
255        int vItemsWeight = this.aInventory.getItemListWeight() + pItem.getItemWeight();
256        if ( vItemsWeight > this.getMaxWeight() ) return false;
257        return true;
258    } // canBeTake()

268    /**
269     * Procédure qui vérifie que l'item associé au nom passer en paramètre peut
270     * être utiliser et réalise l'action que l'item es censé faire
271     * @param pItemName le nom de l'item que le joueur veut utiliser
272     */
273    public void use( final String pItemName )
274    {
275        pkg_item.Item vItem = this.getInventoryItem(pItemName);
276
277        if ( vItem != null ){
278            if ( pItemName.equals("Parchemin") ){
279                this.doubleMaxWeight();
280                this.removeInventoryItem(pItemName);
281                this.aGui.println("Le Parchemin entre en résonnance avec votre Bracelet de Stockage."
282                    + "\nUne lueur vous aveugle ! La capacité de votre Bracelet viens d'être doubler !!");
283                //this.aGui.println(this.inventory());
284            } else if ( pItemName.equals("Rune") ){
285                pkg_item.Beamer vBeamer = (pkg_item.Beamer)(vItem);
286                if ( this.getCurrentRoom().equals(vBeamer.getBeamerRoom()) ){
287                    this.aGui.println("Vous devez effectuez un déplacement avant de vous téléporter au risque de perdre définitivement la Rune");
288                    return;
289                }
290                this.fireBeamer(vBeamer);
291                this.aGui.println("Vous avez été téléporter dans la pièce mémorisé lors du chargement." 
292                    + "\nLa Rune de téléportation perd de son éclat, elle vient de se briser !!!");
293                //this.aGui.println(this.inventory());
294            } else if ( pItemName.equals("Talisman") ){
295                if ( vItem instanceof pkg_item.Beamer ){
296                    pkg_item.Beamer vBeamer = (pkg_item.Beamer)(vItem);
297                    this.fireBeamer(vBeamer);
298                    this.aGui.println("Vous avez été téléporter vers le lieu choisi par le Créateur."
299                        + "\nLe Talisman de téléportation s'éfface de la réalité.");
300                }  else {
301                    this.aGui.println("Vous possedez une version incomplète du Talisman : le Plastron."
302                        + "\nRéunissez les pièces manquantes et assemblez le Talisman pour obtenir une version complète.");
303                }
304            }else {
305                this.aGui.println("Vous pouvez seulement utiliser un téléporteur ou un item d'augmentation de stockage");
306            }
307        } else {
308            this.aGui.println("L'item que vous essayer d'utiliser ne se trouve pas dans votre Bracelet.");
309        }
310    } // use(.)

323    /**
324     * Procédure qui permet de charger un téléporteur si le joueur est en possession de l'item
325     * passe en paramètre
326     * @param pBeamerName le nom de l'item à charger
327     */
328    public void chargeBeamer( final String pBeamerName )
329    {
330        pkg_item.Item vItem = this.getInventoryItem(pBeamerName);
331
332        if ( vItem != null ){ //le joueur possède l'item dans son inventaire
333            //le joueur possède un télépoteur
334            if ( vItem instanceof pkg_item.Beamer ){ // vérifie que vItem est une instance de la classe pkg_item.Beamer
335                pkg_item.Beamer vBeamer = (pkg_item.Beamer)(vItem);
336                if ( vBeamer.getBeamerRoom() == null ){
337                    vBeamer.setBeamerRoom(this.getCurrentRoom());
338                    this.aGui.println("Vous avez charger le téléporteur, il a mémoriser la pièce ou vous vous trouvez actuellement !");
339                } else {
340                    if ( vBeamer.getItemName().equals("Talisman") ){
341                        this.aGui.println("Le Talisman de téléportation, ne peut être chargé puisqu'il vous téléportera"
342                            + "\nvers le lieu prédéfini lors de sa création. Il a été charger par le Sorcier Suprême l'ayant fabriqué.");
343                        return;
344                    }
345                    this.aGui.println("L'item " + pBeamerName + " est déjà chargé !");
346                }
347            } else {
348                this.aGui.println("Vous pouvez seulement charger un téléporteur");
349            }
350        } else {
351            this.aGui.println("Vous ne possèdez pas l'item : " + pBeamerName);
352        }
353    } // chargeBeamer(.)

376    /**
377     * Fonction booleénne qui vérifie que le joueur possède les items nécéssaire pour assembler l'item qui 
378     * à pour nom celui passé en paramètre
379     * @param pItemName le nom de l'item à assembler
380     * @return vrai si l'item est assemblé, sinon retorne faux lorsque le joueur ne peux pas assembler l'item
381     */
382    public boolean build( final String pItemName ) 
383    {
384        pkg_item.Item vItem = this.getInventoryItem(pItemName);
385
386        if ( vItem != null ){
387            if ( vItem.getItemName().equals("Plastron") ){
388                if ( this.getInventoryItem("Fiole") != null && this.getInventoryItem("Vanadinite") != null ){
389                    this.removeInventoryItem("Fiole");
390                    this.removeInventoryItem("Vanadinite");
391                    this.removeInventoryItem("Plastron");
392                    this.aGui.println("Vous avez assemblé le Plastron, il est passé de l'état incomplet, à l'état complet.");
393                    this.aGui.println("Vous êtes désormais en possession d'un Talisman de téléportation complet.\nIl est prêt à être utilisé.");
394                    return true;
395                } else {
396                    this.aGui.println("Il vous manque la Fiole d'Énergie Lunaire ou la Vanadinite dans votre inventaire.");
397                    return false;
398                }
399            } else {
400                this.aGui.println("Vous pouvez seulement assemblez un Plastron de téléportation incomplet !");
401                return false;
402            }
403        } else {
404            this.aGui.println("Vous ne possédez pas l'item dans votre inventaire.");
405            return false;
406        }
407    } // build(.)

409    /**
410     * Procédure qui permet d'ouvrir la porte qui se trouve dans la direction passé en paramètre
411     * si il y a une porte
412     * @param pDirection la direction dans laquel on veut ouvrir une porte
413     */
414    public void openDoor( final String pDirection )
415    {
416        pkg_room.pkg_room.Door vDoor = this.getCurrentRoom().getDoor(pDirection); // la porte dans la direction souhaiter
417
418        if ( vDoor != null ){ // si il y a une porte dans la direction demandé
419            pkg_item.Item vKeyDoor = vDoor.getKeyDoor(); // la clé de la porte
420            pkg_item.Item vKeyInventory = this.getInventoryItem(vKeyDoor.getItemName()); // la clé de la porte dans l'inventaire 
421
422            if ( vKeyInventory != null ){ // le joueur possède la clé de la porte dans son inventaire
423                vDoor.setCanBeCrossed(true);
424                this.aGui.println("La porte dans la direction " + "\"" + pDirection + "\"" + " est maintenant dévérouillé.");
425            } else {
426                this.aGui.println("Vous devez récupérer la clé " + "(" + vKeyDoor.getItemName() + ")" + " de la porte pour pouvoir l'ouvrir !");
427            }
428        } else {
429            this.aGui.println("Il n'y a pas de porte vérouillé dans cette direction.");
430        }
431    } // openDoor(.)
