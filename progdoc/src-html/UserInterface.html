047    /**
048     * Constructeur naturel de UserInteface

057    /**
058     * Methode qui affiche du texte
059     * @param pText ligne de texte
060     */
061    public void print( final String pText )
062    {
063        this.aLog.append(pText);
064        this.aLog.setCaretPosition(this.aLog.getDocument().getLength());
065    } // print(.)
066
067    /**
068     * Méthode qui affiche le texte avec un saut de ligne
069     * @param pText ligne de texte
070     */
071    public void println( final String pText )
072    {
073        this.print(pText + "\n");
074    } // println(.)
075
076    /**
077     * Procédure qui permet de montrer une image dans l'interface
078     * @param pImageName le nom de l'image
079     */
080    public void showImage( final String pImageName )
081    {
082        String vImagePath = "Images/" + pImageName; // pour changer le repertoire
083        URL vImageURL = this.getClass().getClassLoader().getResource( vImagePath );
084        if ( vImageURL == null )
085            System.out.println( "Image non trouvée : " + vImagePath );
086        else {
087            ImageIcon vIcon = new ImageIcon( vImageURL );
088            this.aImage.setIcon( vIcon );
089            this.aMyFrame.pack();
090        }
091    } // showImage(.)
092
093    /**
094     * Active ou désactive la saisi dans le champ de saisi.
095     * @param pOnOff boolean qui active ou desactive la saisi
096     */
097    public void enable( final boolean pOnOff )
098    {
099        this.aEntryField.setEditable( pOnOff ); // enable/disable
100        if ( ! pOnOff ) { // disable
101            this.aEntryField.getCaret().setBlinkRate( 0 ); // cursor won't blink
102            this.aEntryField.removeActionListener( this ); // won't react to entry
103        }
104    } // enable(.)

106    /**
107     * Procédure qui crée l'interface graphique de l'utilisateur
108     */
109    private void createGUI()
110    {
111        this.aMyFrame = new JFrame( "R&LT - by 113 Block | Personne73 Sponso Vitality Clément" ); // change the title
112        this.aEntryField = new JTextField( 34 );
113
114        this.aLog = new JTextArea();
115        this.aLog.setEditable( false );
116        this.aListScroller = new JScrollPane( this.aLog );
117        this.aListScroller.setPreferredSize( new Dimension(600, 300) );
118        this.aListScroller.setMinimumSize( new Dimension(100,100) );
119
120        JPanel vPanel = new JPanel();
121        this.aImage = new JLabel();
122
123        //modification de la police
124        this.aLog.setFont(new Font("Calibri", Font.PLAIN, 12));
125
126        //creation d'un panneau pour le bouton
127        JPanel vButtonPanel = new JPanel();
128
129        //création d'un panneau pour l'inventaire/affichage
130        JPanel vAffichagePanel = new JPanel();
131
132        //création d'une zone de texte pour l'inventaire
133        this.aInventory = new JTextArea();
134        this.aInventory.setEditable( false );
135
136        //ajout de la zone de texte de l'inventaire dans le panneau scrollable 
137        this.aInventoryScroller = new JScrollPane( this.aInventory );
138        this.aInventoryScroller.setPreferredSize( new Dimension(600, 300) );
139        this.aInventoryScroller.setMinimumSize( new Dimension(100,100) );
140
141        // création des boutons
142        JButton vHelpButton = new JButton("aide");
143        JButton vQuitButton = new JButton("quitter");
144        JButton vBackButton = new JButton("retour");
145        JButton vInventoryButton = new JButton("inventaire");
146
147        // modification de la couleur des boutons
148        vHelpButton.setBackground( Color.darkGray );
149        vHelpButton.setForeground( Color.yellow );
150        vQuitButton.setBackground( Color.darkGray );
151        vQuitButton.setForeground( Color.magenta );
152        vBackButton.setBackground( Color.darkGray );
153        vBackButton.setForeground( Color.yellow );
154        vInventoryButton.setBackground( Color.darkGray );
155        vInventoryButton.setForeground( Color.yellow );
156
157        // modification de la couleur de l'interface graphique
158        this.aEntryField.setBackground( Color.gray );
159        this.aEntryField.setForeground( Color.white );
160        this.aLog.setBackground( Color.gray );
161        this.aLog.setForeground( Color.white );
162        vPanel.setBackground( Color.gray );
163        vButtonPanel.setBackground( Color.gray );
164
165        vAffichagePanel.setBackground( Color.gray );
166        this.aInventory.setBackground( Color.gray );
167        this.aInventory.setForeground( Color.white );
168
169        this.aButtonTab = new JButton[4];
170        // ajout de bouton dans le tableau
171        this.aButtonTab[0] = vHelpButton;
172        this.aButtonTab[1] = vBackButton;
173        this.aButtonTab[2] = vInventoryButton;
174        this.aButtonTab[3] = vQuitButton;
175
176        // taille des panneaux
177        vPanel.setPreferredSize( new Dimension(700,600) );
178        vButtonPanel.setPreferredSize( new Dimension(90,100) );
179
180        // ajout de composants au panneau de l'interface
181        vPanel.setLayout( new BorderLayout() ); // ==> only five places
182        vPanel.add( this.aImage, BorderLayout.NORTH );
183        vPanel.add( vAffichagePanel, BorderLayout.CENTER );
184        vPanel.add( this.aEntryField, BorderLayout.SOUTH );
185        vPanel.add( vButtonPanel, BorderLayout.WEST );
186
187        //ajout de composants au panneau des boutons
188        //vButtonPanel.setLayout( new BoxLayout(vButtonPanel, BoxLayout.Y_AXIS));
189        vButtonPanel.setLayout( new GridLayout(4, 0) );
190        vButtonPanel.add( vHelpButton, vButtonPanel );
191        vButtonPanel.add( vInventoryButton, vButtonPanel );
192        vButtonPanel.add( vBackButton, vButtonPanel );
193        vButtonPanel.add( vQuitButton, vButtonPanel );
194
195        // ajout de composants au panneau de l'affichage
196        vAffichagePanel.setLayout( new FlowLayout() );
197        vAffichagePanel.add( this.aListScroller );
198        vAffichagePanel.add( this.aInventoryScroller );
199
200        this.aMyFrame.getContentPane().add( vPanel, BorderLayout.CENTER );
201        // add some event listeners to some components
202        this.aEntryField.addActionListener( this );
203        vHelpButton.addActionListener( this );
204        vBackButton.addActionListener( this );
205        vQuitButton.addActionListener( this );
206        vInventoryButton.addActionListener( this );
207
208        // to end program when window is closed
209        this.aMyFrame.addWindowListener( new WindowAdapter() {
210                public void windowClosing(WindowEvent e) { System.exit(0); }
211            } );
212
213        this.aMyFrame.pack();
214        this.aMyFrame.setVisible( true );
215        this.aEntryField.requestFocus();
216        this.visibilityInventory(false); // rend le panneau d'inventaire invisible par défaut
217        this.setEnableInventoryB(false); // rend le bouton d'inventaire inutilisable par défaut
218        this.setEnableBackB(false); // rend le bouton de retour inutilisable par défaut
219    } // createGUI()

283    /**
284     * Porcédure qui vérifie le type de l'action passé en paramètre et réagis 
285     * en conséquence en fonction de s'il s'agit d'un clic ou d'un appuie 
286     * sur 'entré' dans le clavier
287     * @param pE l'action a vérifié
288     */
289    public void actionPerformed( final ActionEvent pE ) 
290    {
291        // no need to check the type of action at the moment
292        // because there is only one possible action (text input) :
293        for (int i=0; i<this.aButtonTab.length; i++) {
294            if ( pE.getSource() == this.aButtonTab[i] ){
295                this.aEngine.interpretCommand(  pE.getActionCommand() );
296                return;
297            } 
298        }
299        this.processCommand(); // never suppress this line
300    } // actionPerformed(.)
301
302    /**
303     * Procédure qui lis la commande et l'excécute 
304     */
305    private void processCommand()
306    {
307        String vInput = this.aEntryField.getText();
308        this.aEntryField.setText( "" );
309
310        this.aEngine.interpretCommand( vInput );
311    } // processCommand()
